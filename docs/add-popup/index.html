Add Popup

Add Popup to the map.

map.on('load', () => {
    // Add Popup
    const popup = new bkoigl.Popup({ focusAfterOpen: false })
        .setLngLat([ 90.3938010872331, 23.821600277500405 ])
        .setHTML('<h3>Barikoi HQ</h3>')
        .addTo(map)
})
